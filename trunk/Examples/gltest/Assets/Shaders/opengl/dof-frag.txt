// dof-frag.txt

varying vec2 UV;
uniform sampler2D diffuseSampler;

void main()
{
  vec4 diffuse1 = texture2D(diffuseSampler, UV + vec2(0.01, 0.01)).rgba;
  vec4 diffuse2 = texture2D(diffuseSampler, UV).abgr;
//  vec4 diffuse = vec4(UV.x, UV.y, 1, 1);
  gl_FragColor = diffuse1 * 0.5 + diffuse2 * 0.5;
}
