// dof-frag.txt

varying vec2 UV;
uniform sampler2D diffuseSampler;
uniform sampler2D depthSampler;

void main()
{
  if (UV.x < 0.5)
  {
    vec4 diffuse1 = texture2D(diffuseSampler, UV);
    gl_FragColor = diffuse1;
  }
  else
  {
    vec4 depth = texture2D(depthSampler, UV);
    gl_FragColor = depth;
  }
  return;
}
